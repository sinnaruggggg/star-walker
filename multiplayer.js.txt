async function mpGetOtherPlayers() {
    try {
        const response = await fetch(MP_API_URL + '?action=get_players&user_id=' + mpUserId);
        const result = await response.json();
        
        if (result.success) {
            updateOtherPlayers(result.players);
            document.getElementById('mp-online').textContent = 
                `ğŸŸ¢ ì ‘ì†ì: ${result.count + 1}ëª…`;
        }
    } catch (e) {
        console.log('MP í”Œë ˆì´ì–´ ì¡°íšŒ ì‹¤íŒ¨:', e);
    }

function updateOtherPlayers(players) {
    const activeIds = new Set();
    
    players.forEach(player => {
        activeIds.add(player.user_id);
        
        if (!mpOtherPlayers[player.user_id]) {
            createOtherPlayerShip(player);
        } else {
            const ship = mpOtherPlayers[player.user_id];
            ship.mesh.position.lerp(
                new THREE.Vector3(
                    parseFloat(player.pos_x),
                    parseFloat(player.pos_y),
                    parseFloat(player.pos_z)
                ), 0.3
            );
            ship.mesh.rotation.set(
                parseFloat(player.rot_x),
                parseFloat(player.rot_y),
                parseFloat(player.rot_z)
            );
            if (ship.label) {
                ship.label.element.textContent = player.nickname;
            }
        }
    });
    
    Object.keys(mpOtherPlayers).forEach(id => {
        if (!activeIds.has(id)) {
            removeOtherPlayer(id);
        }
    });
}

function createOtherPlayerShip(player) {
    const geometry = new THREE.ConeGeometry(0.3, 1, 8);
    geometry.rotateX(Math.PI / 2);
    
    const material = new THREE.MeshBasicMaterial({ 
        color: 0x00ffff,
        transparent: true,
        opacity: 0.8
    });
    
    const mesh = new THREE.Mesh(geometry, material);
    mesh.position.set(
        parseFloat(player.pos_x),
        parseFloat(player.pos_y),
        parseFloat(player.pos_z)
    );
    
    const glowGeometry = new THREE.SphereGeometry(0.5, 16, 16);
    const glowMaterial = new THREE.MeshBasicMaterial({
        color: 0x00ffff,
        transparent: true,
        opacity: 0.3
    });
    const glow = new THREE.Mesh(glowGeometry, glowMaterial);
    mesh.add(glow);
    
    scene.add(mesh);
    
    const labelDiv = document.createElement('div');
    labelDiv.className = 'mp-player-label';
    labelDiv.textContent = player.nickname;
    const label = new CSS2DObject(labelDiv);
    label.position.set(0, 1, 0);
    mesh.add(label);
    
    mpOtherPlayers[player.user_id] = { mesh, label };
}

function removeOtherPlayer(userId) {
    const player = mpOtherPlayers[userId];
    if (player) {
        scene.remove(player.mesh);
        delete mpOtherPlayers[userId];
    }
}

async function sendChat() {
    const input = document.getElementById('mp-chat-input');
    const message = input.value.trim();
    if (!message) return;
    
    try {
        await fetch(MP_API_URL + '?action=send_chat', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                user_id: mpUserId,
                nickname: mpNickname,
                message: message
            })
        });
        input.value = '';
    } catch (e) {
        console.log('ì±„íŒ… ì „ì†¡ ì‹¤íŒ¨:', e);
    }
}

async function getChat() {
    try {
        const response = await fetch(MP_API_URL + '?action=get_chat&last_id=' + mpChatLastId);
        const result = await response.json();
        
        if (result.success && result.messages.length > 0) {
            const container = document.getElementById('mp-chat-container');
            
            result.messages.forEach(msg => {
                const div = document.createElement('div');
                div.className = 'mp-chat-msg';
                div.innerHTML = `<span class="nick">${msg.nickname}:</span> ${msg.message}`;
                container.appendChild(div);
                mpChatLastId = Math.max(mpChatLastId, parseInt(msg.id));
            });
            
            container.scrollTop = container.scrollHeight;
        }
    } catch (e) {
        console.log('ì±„íŒ… ì¡°íšŒ ì‹¤íŒ¨:', e);
    }
}
